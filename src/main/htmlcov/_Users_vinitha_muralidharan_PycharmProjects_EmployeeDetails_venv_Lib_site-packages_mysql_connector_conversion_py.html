378 def row_to_python(self, row, fields): 

379 """Convert a MySQL text result row to Python types 

380 

381 The row argument is a sequence containing text result returned 

382 by a MySQL server. Each value of the row is converted to the 

383 using the field type information in the fields argument. 

384 

385 Returns a tuple. 

386 """ 

387 i = 0 

388 result = [None]*len(fields) 

389 

390 if not self._cache_field_types: 

391 self._cache_field_types = {} 

392 for name, info in FieldType.desc.items(): 

393 try: 

394 self._cache_field_types[info[0]] = getattr( 

395 self, '_{0}_to_python'.format(name)) 

396 except AttributeError: 

397 # We ignore field types which has no method 

398 pass 

399 

400 for field in fields: 

401 field_type = field[1] 

402 

403 if (row[i] == 0 and field_type != FieldType.BIT) or row[i] is None: 

404 # Don't convert NULL value 

405 i += 1 

406 continue 

407 

408 try: 

409 result[i] = self._cache_field_types[field_type](row[i], field) 

410 except KeyError: 

411 # If one type is not defined, we just return the value as str 

412 try: 

413 result[i] = row[i].decode('utf-8') 

414 except UnicodeDecodeError: 

415 result[i] = row[i] 

416 except (ValueError, TypeError) as err: 

417 err.message = "{0} (field {1})".format(str(err), field[0]) 

418 raise 

419 

420 i += 1 

421 

422 return tuple(result)
