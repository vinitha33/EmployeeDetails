43def intread(buf): 

44 """Unpacks the given buffer to an integer""" 

45 try: 

46 if isinstance(buf, int): 

47 return buf 

48 length = len(buf) 

49 if length == 1: 

50 return buf[0] 

51 elif length <= 4: 

52 tmp = buf + b'\x00'*(4-length) 

53 return struct_unpack('<I', tmp)[0] 

54 tmp = buf + b'\x00'*(8-length) 

55 return struct_unpack('<Q', tmp)[0] 

56 except: 

57 raise

174def read_lc_string(buf): 

175 """ 

176 Takes a buffer and reads a length coded string from the start. 

177 

178 This is how Length coded strings work 

179 

180 If the string is 250 bytes long or smaller, then it looks like this: 

181 

182 <-- 1b --> 

183 +----------+------------------------- 

184 | length | a string goes here 

185 +----------+------------------------- 

186 

187 If the string is bigger than 250, then it looks like this: 

188 

189 <- 1b -><- 2/3/8 -> 

190 +------+-----------+------------------------- 

191 | type | length | a string goes here 

192 +------+-----------+------------------------- 

193 

194 if type == \xfc: 

195 length is code in next 2 bytes 

196 elif type == \xfd: 

197 length is code in next 3 bytes 

198 elif type == \xfe: 

199 length is code in next 8 bytes 

200 

201 NULL has a special value. If the buffer starts with \xfb then 

202 it's a NULL and we return None as value. 

203 

204 Returns a tuple (trucated buffer, bytes). 

205 """ 

206 if buf[0] == 251: # \xfb 

207 # NULL value 

208 return (buf[1:], None) 

209 

210 length = lsize = 0 

211 fst = buf[0] 

212 

213 if fst <= 250: # \xFA 

214 length = fst 

215 return (buf[1 + length:], buf[1:length + 1]) 

216 elif fst == 252: 

217 lsize = 2 

218 elif fst == 253: 

219 lsize = 3 

220 if fst == 254: 

221 lsize = 8 

222 

223 length = intread(buf[1:lsize + 1]) 

224 return (buf[lsize + length + 1:], buf[lsize + 1:length + lsize + 1])

303def read_lc_int(buf): 

304 """ 

305 Takes a buffer and reads an length code string from the start. 

306 

307 Returns a tuple with buffer less the integer and the integer read. 

308 """ 

309 if not buf: 

310 raise ValueError("Empty buffer.") 

311 

312 lcbyte = buf[0] 

313 if lcbyte == 251: 

314 return (buf[1:], None) 

315 elif lcbyte < 251: 

316 return (buf[1:], int(lcbyte)) 

317 elif lcbyte == 252: 

318 return (buf[3:], struct_unpack('<xH', buf[0:3])[0]) 

319 elif lcbyte == 253: 

320 return (buf[4:], struct_unpack('<I', buf[1:4] + b'\x00')[0]) 

321 elif lcbyte == 254: 

322 return (buf[9:], struct_unpack('<xQ', buf[0:9])[0]) 

323 else: 

324 raise ValueError("Failed reading length encoded integer")

349def _parse_os_release(): 

350 """Parse the contents of /etc/os-release file. 

351 

352 Returns: 

353 A dictionary containing release information. 

354 """ 

355 distro = {} 

356 os_release_file = os.path.join("/etc", "os-release") 

357 if not os.path.exists(os_release_file): 

358 return distro 

359 with open(os_release_file) as file_obj: 

360 for line in file_obj: 

361 key_value = line.split("=") 

362 if len(key_value) != 2: 

363 continue 

364 key = key_value[0].lower() 

365 value = key_value[1].rstrip("\n").strip('"') 

366 distro[key] = value 

367 return distro
